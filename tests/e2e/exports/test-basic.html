use serde::{Deserialize, Serialize};
use std::time::{Duration, SystemTime, UNIX_EPOCH};
use uuid::Uuid;

#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct AccessClaims {
    pub sub: String,
    pub exp: usize,
    pub iat: usize,
    pub jti: String,  // JWT ID for revocation
}

#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct RefreshClaims {
    pub sub: String,
    pub exp: usize,
    pub iat: usize,
    pub jti: String,
    pub family: String,  // Token family for rotation detection
}

pub struct TokenPair {
    pub access_token: String,
    pub refresh_token: String,
    pub expires_in: u64,
}

const ACCESS_TOKEN_DURATION: u64 = 15 * 60;  // 15 minutes
const REFRESH_TOKEN_DURATION: u64 = 7 * 24 * 60 * 60;  // 7 days

pub fn create_token_pair(
    user_id: &str,
    access_secret: &[u8],
    refresh_secret: &[u8],
) -> Result<TokenPair, jsonwebtoken::errors::Error> {
    let now = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs() as usize;
    let family = Uuid::new_v4().to_string();
    
    let access_claims = AccessClaims {
        sub: user_id.to_string(),
        exp: now + ACCESS_TOKEN_DURATION as usize,
        iat: now,
        jti: Uuid::new_v4().to_string(),
    };
    
    let refresh_claims = RefreshClaims {
        sub: user_id.to_string(),
        exp: now + REFRESH_TOKEN_DURATION as usize,
        iat: now,
        jti: Uuid::new_v4().to_string(),
        family,
    };
    
    Ok(TokenPair {
        access_token: encode(&Header::default(), &access_claims, &EncodingKey::from_secret(access_secret))?,
        refresh_token: encode(&Header::default(), &refresh_claims, &EncodingKey::from_secret(refresh_secret))?,
        expires_in: ACCESS_TOKEN_DURATION,
    })
}

use std::collections::HashMap;
use std::sync::RwLock;

/// In-memory store for token families (use Redis in production)
pub struct TokenStore {
    /// Maps family_id -> (user_id, latest_jti, is_revoked)
    families: RwLock<HashMap<String, (String, String, bool)>>,
}

impl TokenStore {
    pub fn new() -> Self {
        Self {
            families: RwLock::new(HashMap::new()),
        }
    }
    
    /// Register a new token family
    pub fn register_family(&self, family: &str, user_id: &str, jti: &str) {
        let mut families = self.families.write().unwrap();
        families.insert(family.to_string(), (user_id.to_string(), jti.to_string(), false));
    }
    
    /// Validate and rotate a refresh token
    /// Returns Ok(new_jti) if valid, Err if reuse detected
    pub fn rotate(&self, family: &str, presented_jti: &str, new_jti: &str) -> Result<(), TokenError> {
        let mut families = self.families.write().unwrap();
        
        match families.get_mut(family) {
            Some((_, latest_jti, is_revoked)) => {
                if *is_revoked {
                    return Err(TokenError::FamilyRevoked);
                }
                
                if latest_jti != presented_jti {
                    // REUSE DETECTED! Revoke entire family
                    *is_revoked = true;
                    return Err(TokenError::TokenReuse);
                }
                
                // Valid rotation - update to new JTI
                *latest_jti = new_jti.to_string();
                Ok(())
            }
            None => Err(TokenError::FamilyNotFound),
        }
    }
    
    /// Revoke all tokens for a user (logout)
    pub fn revoke_user(&self, user_id: &str) {
        let mut families = self.families.write().unwrap();
        for (uid, _, is_revoked) in families.values_mut() {
            if uid == user_id {
                *is_revoked = true;
            }
        }
    }
}

#[derive(Debug)]
pub enum TokenError {
    FamilyRevoked,
    TokenReuse,
    FamilyNotFound,
    Expired,
    Invalid,
}

The key security features:

1. **Token Family Tracking**: Each refresh token belongs to a "family" - a chain of rotated tokens

2. **Reuse Detection**: If an old refresh token is used, we know it was stolen (the legitimate user already rotated it)

3. **Family Revocation**: When reuse is detected, we revoke the ENTIRE family, forcing re-authentication

4. **Atomic Rotation**: Each refresh gives a new token and invalidates the old one

use redis::{AsyncCommands, Client};
use std::time::Duration;

pub struct RedisTokenStore {
    client: Client,
    prefix: String,
}

impl RedisTokenStore {
    pub fn new(redis_url: &str) -> Result<Self, redis::RedisError> {
        Ok(Self {
            client: Client::open(redis_url)?,
            prefix: "token_family:".to_string(),
        })
    }
    
    fn family_key(&self, family: &str) -> String {
        format!("{}{}", self.prefix, family)
    }
    
    pub async fn register_family(
        &self,
        family: &str,
        user_id: &str,
        jti: &str,
        ttl: Duration,
    ) -> Result<(), redis::RedisError> {
        let mut conn = self.client.get_multiplexed_async_connection().await?;
        let key = self.family_key(family);
        
        // Store as hash: user_id, latest_jti, revoked
        redis::pipe()
            .hset(&key, "user_id", user_id)
            .hset(&key, "latest_jti", jti)
            .hset(&key, "revoked", "false")
            .expire(&key, ttl.as_secs() as i64)
            .query_async(&mut conn)
            .await?;
        
        Ok(())
    }
    
    pub async fn rotate(
        &self,
        family: &str,
        presented_jti: &str,
        new_jti: &str,
    ) -> Result<(), TokenError> {
        let mut conn = self.client.get_multiplexed_async_connection().await
            .map_err(|_| TokenError::StorageError)?;
        
        let key = self.family_key(family);
        
        // Atomic check-and-update with Lua script
        let script = redis::Script::new(r#"
            local revoked = redis.call('HGET', KEYS[1], 'revoked')
            if revoked == 'true' then
                return 'REVOKED'
            end
            
            local current_jti = redis.call('HGET', KEYS[1], 'latest_jti')
            if current_jti ~= ARGV[1] then
                -- Reuse detected! Revoke family
                redis.call('HSET', KEYS[1], 'revoked', 'true')
                return 'REUSE'
            end
            
            -- Valid rotation
            redis.call('HSET', KEYS[1], 'latest_jti', ARGV[2])
            return 'OK'
        "#);
        
        let result: String = script
            .key(&key)
            .arg(presented_jti)
            .arg(new_jti)
            .invoke_async(&mut conn)
            .await
            .map_err(|_| TokenError::StorageError)?;
        
        match result.as_str() {
            "OK" => Ok(()),
            "REVOKED" => Err(TokenError::FamilyRevoked),
            "REUSE" => Err(TokenError::TokenReuse),
            _ => Err(TokenError::StorageError),
        }
    }
}

The Lua script ensures atomicity - no race conditions between check and update!
