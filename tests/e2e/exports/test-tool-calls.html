// src/services/user/AuthService.ts
import { hash, verify } from 'bcrypt';
import { sign, verify as jwtVerify, JwtPayload } from 'jsonwebtoken';
import { UserRepository } from './UserRepository';
import { EmailService } from '../email_service';
import { Logger } from '../../utils/logger';
import { User } from './types';

export interface AuthTokenPayload extends JwtPayload {
  userId: string;
  role: string;
  type?: 'access' | 'reset';
}

export interface LoginResult {
  user: User;
  accessToken: string;
  refreshToken: string;
  expiresIn: number;
}

export class AuthService {
  private readonly SALT_ROUNDS = 12;
  private readonly ACCESS_TOKEN_EXPIRY = '15m';
  private readonly REFRESH_TOKEN_EXPIRY = '7d';
  private readonly RESET_TOKEN_EXPIRY = '1h';

  constructor(
    private userRepo: UserRepository,
    private emailService: EmailService,
    private logger: Logger,
    private jwtSecret: string
  ) {}

  async hashPassword(password: string): Promise<string> {
    return hash(password, this.SALT_ROUNDS);
  }

  async verifyPassword(password: string, hash: string): Promise<boolean> {
    return verify(password, hash);
  }

  async login(email: string, password: string): Promise<LoginResult> {
    const user = await this.userRepo.findByEmail(email);
    if (!user) {
      this.logger.warn(`Login attempt for unknown email: ${email}`);
      throw new AuthenticationError('Invalid credentials');
    }

    const isValid = await this.verifyPassword(password, user.passwordHash);
    if (!isValid) {
      this.logger.warn(`Invalid password for user: ${user.id}`);
      throw new AuthenticationError('Invalid credentials');
    }

    const accessToken = this.createAccessToken(user);
    const refreshToken = this.createRefreshToken(user);

    await this.userRepo.update(user.id, { lastLoginAt: new Date() });
    this.logger.info(`User logged in: ${user.id}`);

    return {
      user,
      accessToken,
      refreshToken,
      expiresIn: 900, // 15 minutes in seconds
    };
  }

  private createAccessToken(user: User): string {
    return sign(
      { userId: user.id, role: user.role, type: 'access' },
      this.jwtSecret,
      { expiresIn: this.ACCESS_TOKEN_EXPIRY }
    );
  }

  private createRefreshToken(user: User): string {
    return sign(
      { userId: user.id, role: user.role, type: 'refresh' },
      this.jwtSecret,
      { expiresIn: this.REFRESH_TOKEN_EXPIRY }
    );
  }

  async verifyToken(token: string): Promise<AuthTokenPayload> {
    try {
      return jwtVerify(token, this.jwtSecret) as AuthTokenPayload;
    } catch (error) {
      throw new AuthenticationError('Invalid or expired token');
    }
  }

  async changePassword(
    userId: string,
    oldPassword: string,
    newPassword: string
  ): Promise<void> {
    const user = await this.userRepo.findById(userId);
    if (!user) throw new NotFoundError('User not found');

    const isValid = await this.verifyPassword(oldPassword, user.passwordHash);
    if (!isValid) throw new AuthenticationError('Invalid password');

    const passwordHash = await this.hashPassword(newPassword);
    await this.userRepo.update(userId, { passwordHash });
    await this.emailService.sendPasswordChangedEmail(user.email);
    
    this.logger.info(`Password changed for user: ${userId}`);
  }

  async requestPasswordReset(email: string): Promise<void> {
    const user = await this.userRepo.findByEmail(email);
    if (!user) {
      // Don't reveal whether email exists
      this.logger.info(`Password reset requested for unknown email`);
      return;
    }

    const token = sign(
      { userId: user.id, type: 'reset' },
      this.jwtSecret,
      { expiresIn: this.RESET_TOKEN_EXPIRY }
    );

    await this.emailService.sendPasswordResetEmail(user.email, token);
    this.logger.info(`Password reset email sent to user: ${user.id}`);
  }

  async resetPassword(token: string, newPassword: string): Promise<void> {
    const payload = await this.verifyToken(token);
    if (payload.type !== 'reset') {
      throw new AuthenticationError('Invalid reset token');
    }

    const passwordHash = await this.hashPassword(newPassword);
    await this.userRepo.update(payload.userId, { passwordHash });
    
    this.logger.info(`Password reset completed for user: ${payload.userId}`);
  }
}

export class AuthenticationError extends Error {
  constructor(message: string) {
    super(message);
    this.name = 'AuthenticationError';
  }
}

export class NotFoundError extends Error {
  constructor(message: string) {
    super(message);
    this.name = 'NotFoundError';
  }
}

// src/services/user/AuthorizationService.ts
import { UserRepository } from './UserRepository';
import { User, UserRole } from './types';
import { Logger } from '../../utils/logger';

// Permission definitions
export type Permission = 
  | 'users:read'
  | 'users:write'
  | 'users:delete'
  | 'posts:read'
  | 'posts:write'
  | 'posts:delete'
  | 'posts:moderate'
  | 'admin:access'
  | 'analytics:view';

// Role -> Permission mapping
const ROLE_PERMISSIONS: Record<UserRole, Permission[]> = {
  user: [
    'users:read',
    'posts:read',
    'posts:write',
  ],
  moderator: [
    'users:read',
    'posts:read',
    'posts:write',
    'posts:moderate',
    'posts:delete',
  ],
  admin: [
    'users:read',
    'users:write',
    'users:delete',
    'posts:read',
    'posts:write',
    'posts:delete',
    'posts:moderate',
    'admin:access',
    'analytics:view',
  ],
};

export class AuthorizationService {
  constructor(
    private userRepo: UserRepository,
    private logger: Logger
  ) {}

  /**
   * Check if a user has a specific permission
   */
  async hasPermission(userId: string, permission: Permission): Promise<boolean> {
    const user = await this.userRepo.findById(userId);
    if (!user) return false;

    const permissions = ROLE_PERMISSIONS[user.role] || [];
    return permissions.includes(permission);
  }

  /**
   * Check multiple permissions (all must be granted)
   */
  async hasAllPermissions(userId: string, permissions: Permission[]): Promise<boolean> {
    const user = await this.userRepo.findById(userId);
    if (!user) return false;

    const userPermissions = ROLE_PERMISSIONS[user.role] || [];
    return permissions.every(p => userPermissions.includes(p));
  }

  /**
   * Check if user has any of the specified permissions
   */
  async hasAnyPermission(userId: string, permissions: Permission[]): Promise<boolean> {
    const user = await this.userRepo.findById(userId);
    if (!user) return false;

    const userPermissions = ROLE_PERMISSIONS[user.role] || [];
    return permissions.some(p => userPermissions.includes(p));
  }

  /**
   * Require permission or throw
   */
  async requirePermission(userId: string, permission: Permission): Promise<void> {
    const hasPermission = await this.hasPermission(userId, permission);
    if (!hasPermission) {
      this.logger.warn(`Permission denied: ${userId} lacks ${permission}`);
      throw new ForbiddenError(`Missing permission: ${permission}`);
    }
  }

  /**
   * Check role helpers
   */
  async isAdmin(userId: string): Promise<boolean> {
    const user = await this.userRepo.findById(userId);
    return user?.role === 'admin';
  }

  async isModerator(userId: string): Promise<boolean> {
    const user = await this.userRepo.findById(userId);
    return user?.role === 'moderator' || user?.role === 'admin';
  }

  /**
   * Assign role (admin only operation)
   */
  async assignRole(
    adminUserId: string,
    targetUserId: string,
    newRole: UserRole
  ): Promise<User> {
    // Verify admin has permission
    await this.requirePermission(adminUserId, 'users:write');

    // Prevent self-demotion from admin
    if (adminUserId === targetUserId && newRole !== 'admin') {
      throw new ForbiddenError('Cannot demote yourself');
    }

    const updated = await this.userRepo.update(targetUserId, { role: newRole });
    this.logger.info(
      `Role changed: ${targetUserId} -> ${newRole} (by ${adminUserId})`
    );

    return updated;
  }

  /**
   * Get all permissions for a role
   */
  getPermissionsForRole(role: UserRole): Permission[] {
    return [...(ROLE_PERMISSIONS[role] || [])];
  }
}

export class ForbiddenError extends Error {
  constructor(message: string) {
    super(message);
    this.name = 'ForbiddenError';
  }
}

// src/container.ts
import { Database } from './db';
import { CacheService } from './services/cache_service';
import { EmailService } from './services/email_service';
import { Logger } from './utils/logger';

// User domain
import { UserRepository } from './services/user/UserRepository';
import { AuthService } from './services/user/AuthService';
import { AuthorizationService } from './services/user/AuthorizationService';
import { PreferencesService } from './services/user/PreferencesService';
import { UserQueryService } from './services/user/UserQueryService';

export interface Container {
  // Infrastructure
  db: Database;
  cache: CacheService;
  email: EmailService;
  logger: Logger;

  // User domain
  userRepo: UserRepository;
  authService: AuthService;
  authzService: AuthorizationService;
  preferencesService: PreferencesService;
  userQueryService: UserQueryService;
}

export function createContainer(config: AppConfig): Container {
  // Infrastructure (order matters for dependencies)
  const logger = new Logger(config.logLevel);
  const db = new Database(config.databaseUrl);
  const cache = new CacheService(config.redisUrl);
  const email = new EmailService(config.smtpConfig);

  // Repositories
  const userRepo = new UserRepository(db, cache, logger);

  // Services (inject their dependencies)
  const authService = new AuthService(
    userRepo,
    email,
    logger,
    config.jwtSecret
  );

  const authzService = new AuthorizationService(userRepo, logger);
  
  const preferencesService = new PreferencesService(userRepo, logger);
  
  const userQueryService = new UserQueryService(db, cache, logger);

  return {
    db,
    cache,
    email,
    logger,
    userRepo,
    authService,
    authzService,
    preferencesService,
    userQueryService,
  };
}

// For testing - allows partial mocking
export function createTestContainer(
  overrides: Partial<Container> = {}
): Container {
  const defaults = createContainer(testConfig);
  return { ...defaults, ...overrides };
}

Accepted

The UserService class grew to 200+ lines handling multiple responsibilities:

- `AuthService` - Authentication and password management
- `AuthorizationService` - RBAC and permission checks
- `PreferencesService` - User settings management
- `UserQueryService` - Analytics and reporting

- Better separation of concerns

### Negative

- Need to understand dependency graph
- Migration effort for existing code

- Total code increased (more explicit interfaces)
